* Cheat sheet
#+begin_src emacs-lisp
(defun bvoq/_cheatsheet-buffer ()
  (let ((buffer (find-file-noselect (expand-file-name "cheatsheet.org" doom-user-dir))))
    (with-current-buffer buffer
      (when (bound-and-true-p view-mode)
        (view-mode -1))
      (read-only-mode -1))
    buffer))

(defun bvoq/open-cheatsheet ()
  "Open cheatsheet.org from `doom-user-dir' in a temporary window."
  (interactive)
  (let* ((buffer (bvoq/_cheatsheet-buffer))
         (window (get-buffer-window buffer)))
    (if window
        (select-window window)
      (pop-to-buffer buffer '((display-buffer-at-bottom)
                              (window-height . 0.35)
                              (inhibit-same-window . t))))))

(map! :leader :desc "Open cheat sheet" "-" #'bvoq/open-cheatsheet)
#+end_src
